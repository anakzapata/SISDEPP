<script language="javascript">

function buscar(){
	document.ConsultaSimpleEstudianteForm.submit();
}

function agregar(){
	document.ConsultaSimpleEstudianteForm.action = "/sisdepp/prepractica/carga/cargarPaginaBusquedaAyre.do";
	document.ConsultaSimpleEstudianteForm.submit();
}

function eliminar(strCodigo, strSemestre){
	icoError = 'http://tapp.eafit.edu.co/imagenes/v1/icons/ico_error.gif';
	if(esperando == false){
                nombre = document.getElementById(strCodigo).value;
		var objEleccion = confirm('¿Está seguro que desea eliminar el estudiante?' + '\n(' + nombre + ', '+ strCodigo +')' );
		if(objEleccion){
			ocultarError();
			tdNombre = strCodigo;
			strAccion = "eliminar";
			openPage("/sisdepp/prepractica/carga/eliminarEstudiante.do", strCodigo, strSemestre);
		}
	}
}

function aplazar(strCodigo, strSemestre, strEstado){
	icoError = 'http://tapp.eafit.edu.co/imagenes/v1/icons/ico_error.gif';
	if(esperando == false){
		var objEleccion = confirm('¿Está seguro que desea aplazar el estudiante?');
		if(objEleccion){
			ocultarError();
			tdNombre = strCodigo;
			strAccion = "aplazar";
			openPage("/sisdepp/prepractica/estudiantes/cambiarEstadoEstudiante.do", strCodigo,strSemestre, 'APL', strEstado);
		}
	}
}

function llegaRespuesta(){
	if(_getParametro(resultado, 1) == "exito"){
		if(strAccion == "eliminar") hacerEliminar();
                if(strAccion == "aplazar") hacerAplazar();
	}else{
		mostrarError(ajaxerror, icoError, "msgError");
	}
}

function hacerEliminar(){
	hideElement(tdNombre);	
	mostrarError("Estudiante eliminado con éxito", "http://tapp.eafit.edu.co/imagenes/v1/icons/ico_mensaje.gif", "msgMensaje");
}

function hacerAplazar(){
	hideElement(tdNombre);	
	mostrarError("Estudiante aplazado con éxito", "http://tapp.eafit.edu.co/imagenes/v1/icons/ico_mensaje.gif", "msgMensaje");
}
